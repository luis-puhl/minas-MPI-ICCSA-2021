\begin{algorithm}[h]
{\scriptsize
    \SetAlgoVlined
    % \DontPrintSemicolon
    % \KwIn{sample, ModelSet, params as p}
    % \KwOut{Classified Sample}
    % 
    \KwIn{params as p, ModelSet, Sample Stream}
    \KwOut{Classified Stream as $out$}
    % 
    % \KwData{ UnkownSet = $\emptyset$, ModelSleepSet = $\emptyset$, lastCleanupTime = now() }
    % 
    \SetKwFunction{FMain}{MinasOnline}
    \SetKwProg{FnA}{Function}{:}{}
    \FnA{\FMain{p, ModelSet, SampleStream}}{
        UnkownSet = $\emptyset$, ModelSleepSet = $\emptyset$, lastCleanupTime = now()\;
        \ForEach{ sampe from SampleStream }{
            sample.label = unknown\;
            (distance, cluster) = mearestCluster(sample, ModelSet)\;
            \eIf{distance $<$ cluster.radius}{
                sample.label = cluster.label\;
                % \Return sample\;
            }{
                UnkownSet = UnkownSet $\cup$ sample\;
                \If{sizeOf(UnkownSet) $\geq$ p.k}{
                    % \tcc{Novelty Detection}
                    NoveltyDetection(p, *ModelSet, ModelSleepSet, *UnkownSet)\;
                }
                \If{now() $>$ lastCleanup $+$ p.cleaningWindow}{
                    ModelSet = handleModelSleep(ModelSet, ModelSleepSet)\;
                    UnkownSet = removeOldSamples(UnkownSet, lastCleanup)\;
                }
                % \Return sample\;
            }
            $out \leftarrow$ sample\;
        }
    }
    % \;
    \SetKwFunction{FND}{NoveltyDetection}
    \SetKwProg{FnB}{Function}{:}{}
    \FnB{\FND{p, ModelSet, ModelSleepSet, UnkownSet}}{
        newClusters = clustering(UnkownSet)\;
        \ForEach{cl in newClusters}{
            \lIf{sizeOf(cl.sampleSet) $<$ p.minExamplesPerCluster}{\textbf{continue}}
            (distance, near) = mearestCluster(cl, ModelSet $\cup$ ModelSleepSet)\;
            \eIf{distance $<$ near.radius $\times$ p.noveltyFactor}{
                cl.label = near.label\;
                cl.type = extension\;
            }{
                cl.label = p.noveltyIndex\;
                p.noveltyIndex = p.noveltyIndex $+ 1$\;
                cl.type = novelty\;
            }
            UnkownSet = UnkownSet $\ominus$ cl.sampleSet\;
            ModelSet = ModelSet $\oplus$ cl\;
        }
        % \Return{ModelSet}
    }
}
\label{alg:MINAS}
\caption{Our interpretation of MINAS \cite{faria2013novelty,Faria2016minas,Cassales2019a}}
\end{algorithm}
